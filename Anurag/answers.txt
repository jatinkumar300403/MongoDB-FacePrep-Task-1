ROW_NUMBER() or @rownum variable to get odd rows.
SELECT *
FROM (
 SELECT *, ROW_NUMBER() OVER (ORDER BY WORKER_ID) AS row_num
 FROM Worker
) AS temp
WHERE MOD(row_num, 2) = 1;
